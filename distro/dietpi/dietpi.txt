# DietPi Automation Configuration for Ralphglasses Thin Client
# https://dietpi.com/docs/usage/#dietpi-automation
#
# This file automates DietPi first-run setup for a headless thin client
# that boots into i3 + ralphglasses TUI.

# --- System ---
AUTO_SETUP_LOCALE=en_US.UTF-8
AUTO_SETUP_KEYBOARD_LAYOUT=us
AUTO_SETUP_TIMEZONE=America/Los_Angeles
AUTO_SETUP_NET_ETHERNET_ENABLED=1
AUTO_SETUP_NET_WIFI_ENABLED=0
AUTO_SETUP_DHCP_TO_STATIC=0
AUTO_SETUP_NET_HOSTNAME=ralphglasses
AUTO_SETUP_HEADLESS=0
AUTO_SETUP_AUTOSTART_TARGET_INDEX=0
AUTO_SETUP_AUTOMATED=1
AUTO_SETUP_GLOBAL_PASSWORD=dietpi

# --- Software ---
# 25  = i3 window manager
# 65  = alacritty terminal
# 188 = autorandr (multi-monitor persistence)
# 17  = Git
# 130 = Go (for building ralphglasses)
AUTO_SETUP_INSTALL_SOFTWARE_ID=25
AUTO_SETUP_INSTALL_SOFTWARE_ID=65
AUTO_SETUP_INSTALL_SOFTWARE_ID=17

# --- Display ---
# Use X11 (i3 requires X)
AUTO_SETUP_AUTOSTART_TARGET_INDEX=2

# --- Post-install ---
# The Automation_Custom_Script.sh runs after first boot
# It will:
#   1. Install autorandr via apt
#   2. Build ralphglasses binary
#   3. Configure i3 autostart
#   4. Set up autologin
